-- ============================================================================ -- SmartID CRM – Komplett Supabase-setup v3.0 -- KÖR HELA DETTA SCRIPT I SUPABASE SQL EDITOR -- ============================================================================ -- Ändringar från v2: -- - 1 ljudfil per kund (UNIQUE constraint) -- - Admin + Kundtjänst kan ladda upp ljudfiler -- - Admin + Kundtjänst kan permanent radera (skyddat med bekräftelse i CRM) -- ============================================================================ -- ============================================================================ -- DEL 1: ROLES & USERS -- ============================================================================ CREATE TYPE user_role AS ENUM ('admin', 'kundtjanst'); CREATE TABLE profiles ( id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE, email TEXT NOT NULL, full_name TEXT, role user_role NOT NULL DEFAULT 'kundtjanst', created_at TIMESTAMPTZ DEFAULT NOW() ); CREATE OR REPLACE FUNCTION handle_new_user() RETURNS TRIGGER AS $$ BEGIN INSERT INTO profiles (id, email, role) VALUES (NEW.id, NEW.email, 'kundtjanst'); RETURN NEW; END; $$ LANGUAGE plpgsql SECURITY DEFINER; CREATE TRIGGER on_auth_user_created AFTER INSERT ON auth.users FOR EACH ROW EXECUTE FUNCTION handle_new_user(); CREATE OR REPLACE FUNCTION is_admin() RETURNS BOOLEAN AS $$ SELECT EXISTS ( SELECT 1 FROM profiles WHERE id = auth.uid() AND role = 'admin' ); $$ LANGUAGE sql SECURITY DEFINER STABLE; -- Helper: är användaren inloggad alls (admin eller kundtjanst)? CREATE OR REPLACE FUNCTION is_staff() RETURNS BOOLEAN AS $$ SELECT EXISTS ( SELECT 1 FROM profiles WHERE id = auth.uid() AND role IN ('admin', 'kundtjanst') ); $$ LANGUAGE sql SECURITY DEFINER STABLE; -- ============================================================================ -- DEL 2: SELLERS -- ============================================================================ CREATE TABLE sellers ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), name TEXT NOT NULL, email TEXT, active BOOLEAN DEFAULT TRUE, created_at TIMESTAMPTZ DEFAULT NOW() ); CREATE INDEX idx_sellers_active ON sellers(active); -- ============================================================================ -- DEL 3: CUSTOMERS -- ============================================================================ CREATE TYPE customer_status AS ENUM ('active', 'angrad', 'makulerad', 'raderad'); CREATE TYPE package_type AS ENUM ('singel', 'par', 'familj'); CREATE TYPE package_tier AS ENUM ('basic', 'premium'); CREATE TABLE customers ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), -- Kunddata full_name TEXT NOT NULL, personal_number TEXT, phone TEXT, email TEXT, address TEXT, postal_code TEXT, city TEXT, -- Paket package_type package_type, package_tier package_tier, monthly_price INTEGER, -- Tilläggstjänst (Skydda Min Adress) has_addon BOOLEAN DEFAULT FALSE, addon_price INTEGER, -- Bindning binding_months INTEGER DEFAULT 0, binding_start_date DATE, -- Säljare seller_id UUID REFERENCES sellers(id), is_existing_customer BOOLEAN DEFAULT FALSE, -- Status status customer_status DEFAULT 'active', status_changed_at TIMESTAMPTZ, status_reason TEXT, created_at TIMESTAMPTZ DEFAULT NOW(), updated_at TIMESTAMPTZ DEFAULT NOW(), created_by UUID REFERENCES profiles(id) ); CREATE INDEX idx_customers_status ON customers(status); CREATE INDEX idx_customers_seller ON customers(seller_id); CREATE INDEX idx_customers_phone ON customers(phone); CREATE INDEX idx_customers_email ON customers(email); CREATE INDEX idx_customers_created_at ON customers(created_at); CREATE INDEX idx_customers_status_changed ON customers(status_changed_at) WHERE status != 'active'; CREATE OR REPLACE FUNCTION update_updated_at() RETURNS TRIGGER AS $$ BEGIN NEW.updated_at = NOW(); RETURN NEW; END; $$ LANGUAGE plpgsql; CREATE TRIGGER set_updated_at BEFORE UPDATE ON customers FOR EACH ROW EXECUTE FUNCTION update_updated_at(); -- ============================================================================ -- DEL 4: NOTES -- ============================================================================ CREATE TABLE notes ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), customer_id UUID NOT NULL REFERENCES customers(id) ON DELETE CASCADE, content TEXT NOT NULL, created_by UUID REFERENCES profiles(id), created_by_name TEXT, created_at TIMESTAMPTZ DEFAULT NOW() ); CREATE INDEX idx_notes_customer ON notes(customer_id); CREATE INDEX idx_notes_created ON notes(created_at); -- ============================================================================ -- DEL 5: CHANGELOG -- ============================================================================ CREATE TABLE changelog ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), customer_id UUID REFERENCES customers(id) ON DELETE SET NULL, customer_name_snapshot TEXT, action TEXT NOT NULL, field_name TEXT, old_value TEXT, new_value TEXT, description TEXT, changed_by UUID REFERENCES profiles(id), changed_by_name TEXT, changed_at TIMESTAMPTZ DEFAULT NOW() ); CREATE INDEX idx_changelog_customer ON changelog(customer_id); CREATE INDEX idx_changelog_changed_at ON changelog(changed_at); -- ============================================================================ -- DEL 6: AUDIO_FILES (1 per kund + säljutbildning) -- ============================================================================ CREATE TYPE audio_context AS ENUM ('customer', 'training'); CREATE TABLE audio_files ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), context audio_context NOT NULL, customer_id UUID REFERENCES customers(id) ON DELETE CASCADE, title TEXT NOT NULL, storage_path TEXT NOT NULL, file_size INTEGER, mime_type TEXT, uploaded_by UUID REFERENCES profiles(id), created_at TIMESTAMPTZ DEFAULT NOW() ); -- VIKTIGT: 1 ljudfil per kund. Om en ny laddas upp ersätts den gamla via CRM-koden. CREATE UNIQUE INDEX idx_audio_one_per_customer ON audio_files(customer_id) WHERE context = 'customer' AND customer_id IS NOT NULL; CREATE INDEX idx_audio_training ON audio_files(context) WHERE context = 'training';
